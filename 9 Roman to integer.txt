class Solution(object):
    def romanToInt(self, s):
        """
        :type s: str
        :rtype: int
        """
        dict = {'I':1, 'V':5,'X':10, 'L':50, 'C':100, 'D':500, 'M':1000}
        res=0
        size=len(s)
        for i in s:
            res+=dict[i]
        for j in range(size-1):
            if dict[s[j]]<dict[s[j+1]]:
                res-= 2*dict[s[j]]
        return res
