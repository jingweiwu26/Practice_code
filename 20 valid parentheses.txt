class Solution(object):
    def isValid(self, s):
        """
        :type s: str
        :rtype: bool
        """
        stack=[]
        brackets = {'(':')','[':']','{':'}'}
        for i in s:
            if i in brackets:
                stack.append(i)
            elif i in brackets.values():
                if stack:
                    if i!=brackets[stack.pop()]:
                        return False
                elif not stack:
                    return False
        if not stack:
            return True
        else:
            return False
            
class Solution(object):
    def isValid(self, s):
        """
        :type s: str
        :rtype: bool
        """
        dic = {'(':')',
               '[':']',
               '{':'}'}
        stack = []
        for p in s:
            if p in dic:
                stack.append(p)
            elif stack:
                if p != dic[stack.pop()]:
                    return False
            else:
                return False
        if not stack:
            return True   
        return False
