class Solution(object):
    def isValid(self, s):
        """
        :type s: str
        :rtype: bool
        """
        stack=[]
        brackets = {'(':')','[':']','{':'}'}
        for i in s:
            if i in brackets:
                stack.append(i)
            elif i in brackets.values():
                if stack:
                    if i!=brackets[stack.pop()]:
                        return False
                elif not stack:
                    return False
        if not stack:
            return True
        else:
            return False
